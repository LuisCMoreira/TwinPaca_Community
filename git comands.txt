#1: Create a folder for this project on your local hard drive
$ mkdir my-project

#2: change into this folder
$ cd my-project

#3: initialize a new, empty Git repository here
$ git init

...after having written some code + created some files...
#4: add all changes to the next (= first) commit
$ git add .

#5: create this first commit
$ git commit -m "Initial commit"


$git remote add origin <remote repository URL>

$ git push origin master

$git status

//move to branch (create if it does not exist)
git checkout -b 'some name'

//just change (does not create if it does not exist)
git checkout 'some name'

//commit local
git commit -am 'change log description'

//push to new branch
git push  -u origin add-more-instructions

//local branches
git branch -r

//get from githup
git fetch

//get log of repositori
git log

//or to look better
git log --oneline --graph --decorate --all

//exit log
q 

//tag a repo
git tag tag name repo name

//tag old version (v0.1 is tag, 0.1 Release is message, 4417a8f is code from log)
git tag -a v0.1 -m -"0.1 Release" 4417a8f

//see tags
git tags

//push with tags
git push --tags

//create branch
git branch branchname
